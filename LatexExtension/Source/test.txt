\Function{{\tt function}}{{\tt a: array, N: intero}}{{\tt : intero}}
    \State{\tt palo $\gets$ -1}
    \State{\tt dado $\gets$ -1}
    \State{\tt i $\gets$ -1}
    \While{{\tt  i <= N }}
        \If{{\tt  a[i] mod 2 = 0 }}
            \If{{\tt  a[i] > palo }}
                \State{\tt palo $\gets$ a[i]}
            \EndIf%
        \Else%
            \If{{\tt  a[i] > dado }}
                \State{\tt dado $\gets$ a[i]}
            \EndIf%
        \EndIf%
        \State{\tt i $\gets$ i + 1}
    \EndWhile%
    \State{\tt ritorna palo + dado}
\EndFunction%
